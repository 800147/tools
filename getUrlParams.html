getUrlParams.js

Guide

This tool converts url (get) params to given form. 

// getUrlParams(parsers)

const { title, num, isTrue, dayOfWeek } = getUrlParams({
  num: Number,
  isTrue: Boolean,
  dayOfWeek: (param) =>
    (['Sun','Mon','Tue','Wed','Thu','Fri','Sat'])[Number(param)]
});

// ?title=Text&num=5&isTrue=1&dayOfWeek=2
// => { title: 'Text', num: 5, isTrue: true, dayOfWeek: 'Tue' }

// ?num=5&undescribedNum=5
// => { num: 5, undescribedNum: '5' }
// isTrue is undefined here

// ?title=titleA&title=titleB
// => { title: 'titleB' }
